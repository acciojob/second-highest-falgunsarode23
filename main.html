function secondHighest(arr) { if (arr.length < 2) { return -Infinity; } let highest = -Infinity; let secondHighest = -Infinity; for (let num of arr) { if (num > highest) { secondHighest = highest; highest = num; } else if (num > secondHighest && num < highest) { secondHighest = num; } } if (secondHighest === -Infinity) { return -Infinity; } return secondHighest; } // Example usage: console.log(secondHighest([3, 1, 4, 1, 5, 9])); // 5 console.log(secondHighest([1])); // -Infinity console.log(secondHighest([])); // -Infinity console.log(secondHighest([1, 1, 1, 1])); // -Infinity
